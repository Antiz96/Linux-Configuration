## Install YAY & Pamac
$ sudo pacman -S git
$ cd /tmp
$ git clone https://aur.archlinux.org/yay.git
$ cd yay
$ makepkg -si
$ yay -S pamac

## Installing bluetooth support
$ sudo pacman -S bluez pulseaudio-bluetooth
Install bluez-utils-compat (aur) #That package contains deprecated tools that I need to make my USB-Dongle work. If you don't need them, just install bluez-utils via pacman
$ sudo systemctl start bluetooth
$ sudo systemctl enable bluetooth

## Command lines to automatically install my desktop configuration and files configuration ##
/!\Reminder : "/home/rcandau" is hard-coded in "keys" (Lock Screen Shortcut) and in update.desktop (Exec=/home/rcandau/Documents/scripts/update.sh)/!\

#IceWM Conf
$ mkdir -p ~/.icewm && curl https://raw.githubusercontent.com/Antiz96/Linux-Customisation/master/Dotfiles/IceWM/icewm.zip -O && unzip icewm.zip -d ~/.icewm/ && chmod +x ~/.icewm/startup && rm -f icewm.zip
/!\Remember to uncomment the right "Display resolution" command line in the "~/.icewm/startup" script according to the machine (First Line = Desktop | Second Line = Laptop | Third Line = Virtual Machine)/!\

#power.desktop
$ mkdir -p ~/.local/share/applications && curl https://raw.githubusercontent.com/Antiz96/Linux-Customisation/master/Dotfiles/IceWM/power.desktop -o ~/.local/share/applications/power.desktop

#update.desktop
$ mkdir -p ~/.local/share/applications && curl https://raw.githubusercontent.com/Antiz96/Linux-Customisation/master/Dotfiles/IceWM/update.desktop -o ~/.local/share/applications/update.desktop

#update.sh
$ mkdir -p ~/Documents/scripts && curl https://raw.githubusercontent.com/Antiz96/Linux-Customisation/master/Dotfiles/IceWM/update.sh -o ~/Documents/scripts/update.sh && chmod +x ~/Documents/scripts/update.sh

#update_notifier.sh
$ mkdir -p ~/Documents/scripts && curl https://raw.githubusercontent.com/Antiz96/Linux-Customisation/master/Dotfiles/IceWM/update_notifier.sh -o ~/Documents/scripts/update_notifier.sh && chmod +x ~/Documents/scripts/update_notifier.sh

#terminalrc
$ mkdir -p ~/.config/xfce4/terminal && curl https://raw.githubusercontent.com/Antiz96/Linux-Customisation/master/Dotfiles/IceWM/terminalrc -o ~/.config/xfce4/terminal/terminalrc
(Delete comments at the top)

#.tmux.conf
$ curl https://raw.githubusercontent.com/Antiz96/Linux-Customisation/master/Dotfiles/General/tmux.conf -o ~/.tmux.conf
(Delete comments at the top and in the copy/paste section)

#zathurarc
$ mkdir -p ~/.config/zathura/ && curl https://raw.githubusercontent.com/Antiz96/Linux-Customisation/master/Dotfiles/General/zathurarc -o ~/.config/zathura/zathurarc
(Delete comments at the top)

#.bashrc
$ curl https://raw.githubusercontent.com/Antiz96/Linux-Customisation/master/Dotfiles/General/bashrc%20\(Arch%20Based%20Distro\) -o ~/.bashrc && source ~/.bashrc

## Setup the crontab for the update notifier
$ sudo systemctl start cronie && sudo systemctl enable cronie
$ crontab -e
#Update Notifier
*/30 * * * * ~/Documents/scripts/update_notifier.sh

## Reboot to apply changes
$ reboot

## Keyboard Shortcuts (that's just a reminder, those are set in my config files)

Super + A = Open Dmenu App Finder/Launcher
Super + F = Switch size of windows (Maximize/Minimize)
Super + D = Close the window
Super + E = Thunar
Super + C = Start calculator
Super + M = Display the desktop
Super + T = Terminal
Super + L = Lock the screen
Super + V = Tile open windows vertically
