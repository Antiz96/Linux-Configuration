##Debian Server
Just a quick reminder on how I install a minimalistic Debian Server environnement for various needs

#Base Install
Minimal Install (nothing checked)
Partitionning :
550 MB EFI
4G Swap
Left ext4 / (0% reserved block)

#Install sudo and put regular user in the sudo group
$ apt install sudo
$ usermod -aG sudo rcandau
(Log into regular account)

#Install useful packages
$ sudo apt install man bash-completion openssh-server dnsutils traceroute rsync zip unzip diffutils firewalld mlocate htop curl vim openssl telnet (chronyd)

#Enable/start/configure various things
$ sudo systemctl enable --now ssh
$ sudo updatdb
$ sudo mandb
$ sudo systemctl enable --now firewalld
$ sudo firewall-cmd --add-service="ssh" --permanent
$ sudo firewall-cmd --reload

#Install qemu-guest-agent (if needed, for promox)
$ sudo apt install qemu-guest-agent
$ sudo systemctl enable --now qemu-guest-agent

#Downlaod my .bashrc
$ curl https://raw.githubusercontent.com/Antiz96/Linux-Customisation/main/Dotfiles/General/bashrc%20\(Debian-Ubuntu%20Based%20Distro%20SERVER%20-%20WSL\) -o ~/.bashrc
$ vim ~/.bashrc #Get rid of TMUX things and flatpak in the fullupgrade alias

#Static IP Adress
$ sudo vim /etc/network/interfaces
[...]
# The primary network interface
allow-hotplug ens18
iface ens18 inet static
        address 192.168.1.199/24
        gateway 192.168.1.254
        dns-nameservers 192.168.1.1

#Reboot
$ sudo reboot
