##Install Arch Linux on my raspberry PI 4
https://archlinuxarm.org/platforms/armv8/broadcom/raspberry-pi-4

##Update the machine
$ pacman -Syy
$ pacman -Syu

##Activate the swap partition
$ swapon /dev/mmcblk0p3
$ vi /etc/fstab
[...]
/dev/mmcblk0p3  none    swap    defaults        0       0

##Configure Arch Linux according to my preferences (Additional packages and DE just below the link)
https://github.com/Antiz96/Linux-Customisation/blob/master/Arch%20Linux%20Install/Arch%20Linux%20Installation.txt
#Additionnal useful packages, drivers and DE
$ sudo pacman -S base-devel linux-headers man bash-completion xorg-server xf86-video-fbdev xfce4 mousepad ristretto file-roller thunar-archive-plugin xdg-user-dirs-gtk pulseaudio network-manager-applet blueman xfce4-pulseaudio-plugin numlockx gvfs lightdm lightdm-gtk-greeter
$ sudo systemctl enable lightdm

##Install yay, bluetooth support and useful packages
https://github.com/Antiz96/Linux-Customisation/blob/master/Arch%20Linux%20Install/Arch%20Linux%20Post%20Install.txt
$ sudo pacman -S bluez pulseaudio-bluetooth bluez-utils keepassxc vlc chromium tmux dmenu zathura zathura-pdf-poppler mlocate htop neofetch
$ sudo systemctl start bluetooth
$ sudo systemctl enable bluetooth

##Modify the lightdm configuration file to activate numlock at boot
$ sudo vi /etc/lightdm/lightdm.conf
[Seat:*]
[...]
greeter-setup-script=/usr/bin/numlockx on
[...]

## No sound
# If you have no sound, please check that the pulseaudio package is installed on your raspberry
# If it is and you still have no sound, add this to /boot/config.txt (check this link https://archlinuxarm.org/platforms/armv7/broadcom/raspberry-pi-2)
dtparam=audio=on
hdmi_drive=2

## Raspberry Pi uncorrect date/time after boot
# Raspberry Pi has no HWclock, so we'll use a fake one from the AUR
# Set the correct date for your raspberry, then install/start/enable fake-hwclock
$ sudo date -s "yyyy-mm-dd hh:mm:ss"
$ yay -S fake-hwclock 
$ sudo systemctl start fake-hwclock
$ sudo systemctl enable fake-hwclock
