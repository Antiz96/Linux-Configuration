## Install YAY & Pamac
$ sudo pacman -S git
$ cd /tmp
$ git clone https://aur.archlinux.org/yay.git
$ cd yay
$ makepkg -si
$ yay -S pamac

## Installing bluetooth support
$ sudo pacman -S bluez pulseaudio-bluetooth
Install bluez-utils-compat (aur) #That package contains deprecated tools that I need to make my USB-Dongle work. If you don't need them, just install bluez-utils via pacman
$ sudo systemctl start bluetooth
$ sudo systemctl enable bluetooth

## Switch from XFCE compositor to picom (for animation and blur support)
XFCE Settings -> "Window Manager Tweaks" -> "Compositor" -> uncheck "Enable display compositing"
XFCE Settings -> "Session and Startup" -> "Application Autostart" -> add item with the command "picom"
