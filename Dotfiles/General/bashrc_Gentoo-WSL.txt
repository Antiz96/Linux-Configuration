#
# ~/.bashrc
#

# If not running interactively, don't do anything
[[ $- != *i* ]] && return

##Prompt##

#Red if TMUX is not running
if [ -z "${TMUX}" ]; then
        PS1="\[\e[0;31m[\u@\h \W]\$ \e[m\]"
#White if TMUX is running
else
        PS1="[\u@\h \W]\$ "
fi

##Tmux autostart##

[ -x "$(command -v tmux)" ] \
  && [ -z "${TMUX}" ] \
  && (tmux attach || tmux) >/dev/null 2>&1

##Env var##

#Editor
export EDITOR="/usr/bin/vim"

##Alias##

#Ls
alias ls="ls --color=auto"
alias ll="ls -ltr"
alias l="ls -ltrA"

#Grep
alias grep="grep --color=auto"
alias egrep="egrep --color=auto"
alias fgrep="fgrep --color=auto"

#Manipulating files
alias mv="mv -i"
alias cp="cp -i"
alias ln="ln -i"
alias rm="rm -i"

#Editor
alias vi="vim"

#Package manager
alias update="sudo emerge --sync"
alias upgrade="sudo emerge -avuDN --with-bdeps=y @world"
alias fullupgrade="sudo emerge --sync && sudo emerge -avuDN --with-bdeps=y @world"
alias install="sudo emerge -av"
alias remove="sudo emerge -aC"
alias search="emerge -s"
alias pkglist="qlist -I"
alias cleanorphans="sudo emerge -ac"

#Tmux
alias tkill="tmux kill-session -t"
alias tswitch="tmux a -t"

#Sudo
alias sudo="sudo "
