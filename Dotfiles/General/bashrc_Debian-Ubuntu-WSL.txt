#
# ~/.bashrc
#

# If not running interactively, don't do anything
[[ $- != *i* ]] && return

##Prompt##

#Red if TMUX is not running
if [ -z "${TMUX}" ]; then
        PS1="\[\e[0;31m[\u@\h \W]\$ \e[m\]"
#Agnoster bash theme if TMUX is running
else
        export THEME="${HOME}/.bash/themes/agnoster-bash/agnoster.bash"
        if [ -f "${THEME}" ]; then
                source "${THEME}"
        fi
fi

##Tmux autostart##

[ -x "$(command -v tmux)" ] \
  && [ -z "${TMUX}" ] \
  && (tmux attach || tmux) >/dev/null 2>&1

##Env var##

#Editor
export EDITOR="/usr/bin/vim"

##Alias##

#Ls
alias ls="ls --color=auto"
alias ll="ls -ltr"
alias l="ls -ltrA"

#Grep
alias grep="grep --color=auto"
alias egrep="egrep --color=auto"
alias fgrep="fgrep --color=auto"

#Manipulating files
alias mv="mv -i"
alias cp="cp -i"
alias ln="ln -i"
alias rm="rm -i"

#Editor
alias vi="vim"

#Package manager
alias update="sudo apt update"
alias upgrade="sudo apt update && sudo apt upgrade"
alias fullupgrade="sudo apt update && sudo apt full-upgrade"
alias install="sudo apt install"
alias remove="sudo apt remove"
alias purge="sudo apt purge"
alias search="apt search"
alias pkglist="apt list --installed"
alias cleancache="sudo apt clean"
alias cleanorphans="sudo apt autoremove --purge"

#Tmux
alias tkill="tmux kill-session -t"
alias tswitch="tmux a -t"

#Fix for Distrobox/Docker in WSL <-- https://github.com/kubernetes/kubernetes/issues/61058#issuecomment-372764783
alias distrobox-repair="sudo mount --make-rshared /"

#Sudo
alias sudo="sudo "
