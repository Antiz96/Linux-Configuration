#
# ~/.bashrc
#

# If not running interactively, don't do anything
[[ $- != *i* ]] && return

##Prompt##

#Red if TMUX is not running
if [ -z "${TMUX}" ]; then
        PS1="\[\e[0;31m[\u@\h \W]\$ \e[m\]"
#Agnoster bash theme if TMUX is running
else
        export THEME="${HOME}/.bash/themes/agnoster-bash/agnoster.bash"
        if [ -f "${THEME}" ]; then
                source "${THEME}"
        fi
fi

##Tmux autostart##

[ -x "$(command -v tmux)" ] \
  && [ -z "${TMUX}" ] \
  && (tmux attach || tmux) >/dev/null 2>&1

##Env var##

#Editor
export EDITOR="/usr/bin/vim"

##Alias##

#Ls
alias ls="ls --color=auto"
alias ll="ls -ltr"
alias l="ls -ltrA"

#Grep
alias grep="grep --color=auto"
alias egrep="egrep --color=auto"
alias fgrep="fgrep --color=auto"

#Manipulating files
alias mv="mv -i"
alias cp="cp -i"
alias ln="ln -i"
alias rm="rm -i"

#Editor
alias vi="vim"

#Package manager
alias update="dnf check-update"
alias upgrade="sudo dnf upgrade --refresh"
alias fullupgrade="sudo dnf distro-sync --refresh"
alias install="sudo dnf install"
alias remove="sudo dnf remove"
alias search="dnf search"
alias pkglist="dnf list installed"
alias cleancache="sudo dnf clean all"
alias cleanorphans="sudo dnf autoremove"

#Tmux
alias tkill="tmux kill-session -t"
alias tswitch="tmux a -t"

#Sudo
alias sudo="sudo "
